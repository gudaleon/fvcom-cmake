if (ENABLE_OPENMP)
    find_package(OpenMP)

        set(HAVE_OPENMP TRUE)
        set(CMAKE_Fortran_FLAGS "${CMAKE_Fortran_FLAGS} ${OpenMP_C_FLAGS}")
    endif()
endif ()
